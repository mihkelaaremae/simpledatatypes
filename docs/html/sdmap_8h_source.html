754/*
755 * @brief sdmap slot object.
756 *
757 * The values in sdmap_slot have special meaning
758 * height < 0 -> slot is an empty slot
759 * height >= 0 -> slot is a node
760 *
761 * if empty slot:
762 * right -> next empty slot
763 * right == -1 -> no more empty slots
764 *
765 * if node:
766 * left == index_of_self -> no child on the left
767 * right == index_of_self -> no child on the right
768 * parent == -1 -> root node
769 *
770 */
771typedef struct sdmap_slot
772{
773 sdmap_index_t left;
774 sdmap_index_t right;
775 sdmap_index_t parent;
776 int8_t height;
777} sdmap_slot;
